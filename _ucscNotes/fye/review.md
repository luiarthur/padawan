      $$
      \begin{array}{rcl}
        a_0 &=& \ds\frac{2}{L} \int_{x_0}^{x_0+L} f(x) dx\\
        a_n &=& \ds\frac{2}{L} \int_{x_0}^{x_0+L} f(x) \cos\p{\frac{2\pi nx}{L}}dx\\
        b_n &=& \ds\frac{2}{L} \int_{x_0}^{x_0+L} f(x) \sin\p{\frac{2\pi nx}{L}}dx\\
      \end{array}
      $$
